数组去重

对数组进行去重 1、Reduce() 方法 let myArray = ['a', 'b', 'a', 'b', 'c', 'e', 'e', 'c', 'd', 'd', 'd', 'd'] let myArrayWithNoDuplicates = myArray.reduce( (previousValue, currentValue) => { if (previousValue.indexOf(currentValue) === -1) { previousValue.push(currentValue) } return previousValue }, []) console.log(myArrayWithNoDuplicates) //output：[ 'a', 'b', 'c', 'e', 'd' ] 2、利用 Set() 中元素是唯一的特点 arrayWithNoDuplicates = Array.from(new Set(myArray)) console.log(arrayWithNoDuplicates) //output：[ 'a', 'b', 'c', 'e', 'd' ] console.log([...new Set(myArray)]) //output：[ 'a', 'b', 'c', 'e', 'd' ]
